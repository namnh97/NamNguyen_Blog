```
//compute the dot product AB . BC
int dot(int[] A, int[] B, int[] C){
    AB = new int[2];
    BC = new int[2];
    AB[0] = B[0] - A[0]; //xB - xA
    AB[1] = B[1] - A[1]; //yB - yA
    BC[0] = C[0] - B[0]; //xC - xB
    BC[1] = C[1] - B[1]; //yC - yB
    int dot = AB[0] * BC[0] + AB[1] * BC[1];
    return dot;
}
//compute the cross product AB x AC 
int cross(int[] A, int[] B, int[] C){
	AB = new int[2];
	AC = new int[2];
	AB[0] = B[0] - A[0]; // xB - xA
	AB[1] = B[1] - A[1]; // yB - yA
	AC[0] = C[0] - A[0]; // xC - xA
	AC[1] = C[1] - A[1]; // yC - yA
	int cross = AB[0] * AC[1] - AB[1] * AC[0];
}

//Compute the distance from A to B
double distance(int[] A, int[] B){
	int d1 = A[0] - B[0];
	int d2 = A[1] - B[1];
	return sqrt(d1 * d1 + d2 * d2);
}

double linePointDist(int[] A, int[] B, int[] C, boolean isSegment){

	}

}
```

# Polygon area
```
int area = 0;
int N = lengthof(p);
for(int i = 1; i + 1 < N; i++){
	int x1 = p[i][0] - p[0][0];
	int y1 = p[i][1] - p[0][1];
	int x2 = p[i + 1][0] - p[0][0];
	int y2 = p[i + 1][0] - p[0][1];
	int cross = x1 * y2 - x2 * y1;
	area += cross;
}
return abs(area / 2);
```

```
convexHull(point[] X){
	int N = lengthof(X);
	int p = 0;
	//First find the leftmost point
	for(int i = 1; i < N; i++){
		if(X[i] < X[p]){
		 	p = i;
		}
	}
	int start = p;
	do{
		int n = -1;
		for(int i = 0; i < N; i++){
			//Don't go back to the same point you came frome
			if(i == p) continue;

			// if there is no N yet, set it to i
			if(n == -1) n = i;

			if(cross < 0){
				//as described above, set N = X

			}
		}

	}while(start != p);
```
